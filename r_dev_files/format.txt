#sherlock
doyle_books <- gutenberg_works(author == "Doyle, Arthur Conan") %>% gutenberg_download(meta_fields = "title")
unique(doyle_books$title)
Book_1 <- doyle_books %>% filter(title == "The Return of Sherlock Holmes")
writeLines(Book_1$text, "The Return of Sherlock Holmes.txt")
