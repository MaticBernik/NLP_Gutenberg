#sherlock
doyle_books <- gutenberg_works(author == "Doyle, Arthur Conan") %>% gutenberg_download(meta_fields = c("title", "author", "gutenberg_author_id", "language", "gutenberg_bookshelf", "has_text", "rights"))
unique(doyle_books$title)
Book_1 <- doyle_books %>% filter(title == "The Return of Sherlock Holmes")
writeLines(Book_1$text, "The Return of Sherlock Holmes.txt")
